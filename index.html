    <script>

        // --- YOUR GITHUB USERNAME ---

        const GITHUB_USERNAME = 'Permanentfrost';


        // Repository categorizer based on your exact repos

        const categories = {

            "Infrastructure & Homelab": ['ProxmoxTorrentVMSetup', 'proxmox-nextcloud', 'HTTPS_via_Caddy_forNextcloud_LANonly', 'omv-smb-private-public-setup', 'add-external-usb-backup-proxmox', 'HomeLab'],

            "Cybersecurity & Audit": ['az500studycram_checklist', 'ITAuditGuide', 'CyberSecFrameworks', 'sme-cysec-checklist', 'SwissDataprotectionGuide', 'htb_pt_guide'],

            "Hardware & Electronics": ['raspi-pico-cheatsheet', 'HardwareProjects'],

            "Gaming & Optimization": ['wotlk_wow', 'pc_performance']

        };


        function getCategory(repoName) {

            for (const [category, repos] of Object.entries(categories)) {

                if (repos.includes(repoName)) return category;

            }

            return "Other Projects"; // Catches any new repos you make that aren't categorized yet

        }


        async function fetchRepos() {

            try {

                // Fetching from your specific GitHub profile

                const response = await fetch(`https://api.github.com/users/${GITHUB_USERNAME}/repos?per_page=100&sort=updated`);

                if (!response.ok) throw new Error('Failed to fetch');

                const repos = await response.json();


                // Group repos

                const groupedRepos = {};

                repos.forEach(repo => {

                    const cat = getCategory(repo.name);

                    if (!groupedRepos[cat]) groupedRepos[cat] = [];

                    groupedRepos[cat].push(repo);

                });


                renderRepos(groupedRepos);

            } catch (error) {

                document.getElementById('repo-container').innerHTML = `<p class="text-red-400">Error loading repositories. Please check the connection.</p>`;

            }

        }


        function renderRepos(groupedRepos) {

            const container = document.getElementById('repo-container');

            container.innerHTML = '';


            // Render categories in order

            Object.keys(categories).forEach(category => {

                if (!groupedRepos[category] || groupedRepos[category].length === 0) return;


                const section = document.createElement('div');

                

                // Category Header

                section.innerHTML = `

                    <h2 class="text-2xl font-semibold text-white mb-6 border-b border-gray-800 pb-2">${category}</h2>

                    <div class="grid grid-cols-1 md:grid-cols-2 lg:grid-cols-3 gap-6">

                        ${groupedRepos[category].map(repo => `

                            <a href="${repo.html_url}" target="_blank" rel="noopener noreferrer" class="glass-card rounded-xl p-6 flex flex-col h-full cursor-pointer no-underline group">

                                <h3 class="text-xl font-medium text-white mb-3 group-hover:text-violetAccent transition-colors">${repo.name}</h3>

                                <p class="text-sm text-gray-400 mb-6 flex-grow leading-relaxed">${repo.description || 'No description provided.'}</p>

                                <div class="flex items-center justify-between text-xs text-gray-500 font-mono mt-auto">

                                    <span class="flex items-center gap-1">

                                        <svg class="w-4 h-4" fill="currentColor" viewBox="0 0 24 24"><path d="M12 2L15.09 8.26L22 9.27L17 14.14L18.18 21.02L12 17.77L5.82 21.02L7 14.14L2 9.27L8.91 8.26L12 2Z"></path></svg>

                                        ${repo.stargazers_count}

                                    </span>

                                    <span>Updated: ${new Date(repo.updated_at).toLocaleDateString('en-GB')}</span>

                                </div>

                            </a>

                        `).join('')}

                    </div>

                `;

                container.appendChild(section);

            });

            

            // Render "Other Projects" if there are any new ones

            if (groupedRepos["Other Projects"] && groupedRepos["Other Projects"].length > 0) {

                 const otherSection = document.createElement('div');

                 otherSection.innerHTML = `

                    <h2 class="text-2xl font-semibold text-white mb-6 border-b border-gray-800 pb-2 mt-12">Other Projects</h2>

                    <div class="grid grid-cols-1 md:grid-cols-2 lg:grid-cols-3 gap-6">

                        ${groupedRepos["Other Projects"].map(repo => `

                            <a href="${repo.html_url}" target="_blank" rel="noopener noreferrer" class="glass-card rounded-xl p-6 flex flex-col h-full cursor-pointer no-underline group">

                                <h3 class="text-xl font-medium text-white mb-3 group-hover:text-violetAccent transition-colors">${repo.name}</h3>

                                <p class="text-sm text-gray-400 mb-6 flex-grow leading-relaxed">${repo.description || 'No description provided.'}</p>

                                <div class="flex items-center justify-between text-xs text-gray-500 font-mono mt-auto">

                                    <span>Updated: ${new Date(repo.updated_at).toLocaleDateString('en-GB')}</span>

                                </div>

                            </a>

                        `).join('')}

                    </div>

                `;

                container.appendChild(otherSection);

            }

        }


        // Initialize

        fetchRepos();

    </script>
